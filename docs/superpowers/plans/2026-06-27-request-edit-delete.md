# Request Edit/Delete Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add edit/delete actions for requests (gated by `manage_requests` permission) and remove the unique constraint on `requests.code`.

**Architecture:** Extend the IPC contract first, then generate a schema migration, then implement and test the main-process handlers, then update UI (Settings permission label + two new dialogs in RequestDetail). Each task is independently testable. `departmentId` is added to `RequestDetail` so the edit dialog can pre-select the correct department without fragile name-matching.

**Tech Stack:** Electron, better-sqlite3, Drizzle ORM, React 18, TanStack Query, TypeScript, Vitest

## Global Constraints

- All user-facing strings must be in **Vietnamese**.
- Admin role implicitly has every permission (`requirePermission` returns `null` for admins).
- `manage_requests` permission: both admin and permitted staff may edit/delete.
- Button order in RequestDetail header: `[In phiếu]  [Sửa]  [Xoá]  [Thêm thiết bị]` — Sửa/Xoá only visible when `hasPermission('manage_requests')`.
- Error toasts: fixed-position red `div`, `bottom: 24, right: 24, zIndex: 200` — same pattern as the existing toasts in `RequestDetail.tsx`.
- Duplicate request codes are silently allowed after the migration — no warning, no block.
- Run TypeScript check: `npm run typecheck` (runs both node + web sides).
- Run tests: `npx vitest run electron/main/handlers/requests.test.ts` (avoids ABI issue — see CLAUDE.md).

describe('requests.delete', () => {

    const db = freshDb()
    // Insert a request to delete while ADMIN, then switch to STAFF for the call

      code: 'DEL-PERM', departmentId: deptId, employeeId: null,
      createdBy: 1, createdAt: new Date().toISOString(), notes: null,
    }).run()
    const reqId = Number(ins.lastInsertRowid)

    session.current = STAFF_SESSION
    const h = makeRequestHandlers(db)
    const res = await h.delete({ id: reqId })
    expect(res.ok).toBe(false)
    if (!res.ok) expect(res.error.code).toBe('FORBIDDEN')
  })

  it('rejects invalid id', async () => {
    const db = freshDb()
    const h = makeRequestHandlers(db)
    // @ts-expect-error testing bad input
    const res = await h.delete({ id: null })
    expect(res.ok).toBe(false)
  })

    const db = freshDb()
    const h = makeRequestHandlers(db)
    const res = await h.delete({ id: 99999 })
    expect(res.ok).toBe(false)
    if (!res.ok) expect(res.error.code).toBe('NOT_FOUND')
  })

  it('deletes request and its allocations, restores unreturned devices to available', async () => {
    const db = freshDb()

    // Find a device with 'available' status from seed
    const availDev = db.select().from(devices).all().find(d => d.status === 'available')
    if (!availDev) throw new Error('no available device in seed')

    const deptId = db.select({ id: departments.id }).from(departments).all()[0].id
    const now = new Date().toISOString()

    // Create request
    const ins = db.insert(requests).values({
      code: 'DEL-001', departmentId: deptId, employeeId: null,
      createdBy: 1, createdAt: now, notes: null,
    }).run()
    const reqId = Number(ins.lastInsertRowid)

    // Mark device as allocated and create allocation
    db.update(devices).set({ status: 'allocated', updatedAt: now })
      .where(eq(devices.id, availDev.id)).run()
    db.insert(allocations).values({
      requestId: reqId, deviceId: availDev.id, employeeId: null,
      departmentId: deptId, issuedBy: 1, issuedAt: now,
    }).run()

    const h = makeRequestHandlers(db)
    const res = await h.delete({ id: reqId })

    // Request gone
    expect(db.select().from(requests).where(eq(requests.id, reqId)).all()).toHaveLength(0)

    expect(db.select().from(allocations).all().filter(a => a.requestId === reqId)).toHaveLength(0)

    // Device restored to 'available'
    const devRow = db.select().from(devices).where(eq(devices.id, availDev.id)).all()[0]
    expect(devRow?.status).toBe('available')
  })

  it('does not change status of already-returned devices', async () => {
    const db = freshDb()

    const availDev = db.select().from(devices).all().find(d => d.status === 'available')
    if (!availDev) throw new Error('no available device in seed')

      createdBy: 1, createdAt: now, notes: null,
    }).run()

    db.insert(allocations).values({
      requestId: reqId, deviceId: availDev.id, employeeId: null,
      departmentId: deptId, issuedBy: 1, issuedAt: now, returnedAt: now,
    }).run()

    const h = makeRequestHandlers(db)
    await h.delete({ id: reqId })

    // Device status unchanged (still 'available')
    const devRow = db.select().from(devices).where(eq(devices.id, availDev.id)).all()[0]
    expect(devRow?.status).toBe('available')
  })
})

  if (!args?.id || typeof args.id !== 'number') {

  const existing = db.select({ id: requests.id }).from(requests)
    .where(eq(requests.id, args.id)).all()[0]
  if (!existing) {
    return { ok: false, error: { code: 'NOT_FOUND', message: 'Không tìm thấy phiếu đề nghị.' } }
  }

  const now = new Date().toISOString()

  const unreturned = db
    .select({ deviceId: allocations.deviceId })
    .from(allocations)
    .where(and(eq(allocations.requestId, args.id), isNull(allocations.returnedAt)))
    .all()

  for (const alloc of unreturned) {
    if (alloc.deviceId != null) {
      db.update(devices)
        .set({ status: 'available', updatedAt: now })
        .where(eq(devices.id, alloc.deviceId))
        .run()
    }
  }

  db.delete(allocations).where(eq(allocations.requestId, args.id)).run()
  db.delete(requests).where(eq(requests.id, args.id)).run()

  return { ok: true, data: { ok: true } }
},
